      function fmt(n) {

        return '¥' + Math.round(n || 0).toLocaleString();

      }

      function daysUntil(d) {

        return Math.ceil((new Date(d) - new Date()) / 86400000);

      }

      function tripStatus(t) {

        const s = daysUntil(t.startDate),

          e = daysUntil(t.endDate);

        if (s > 0)

          return { label: 'あと' + s + '日', bg: '#EEEDFE', color: '#3C3489' };

        if (e >= 0)

          return { label: '旅行中！', bg: '#E1F5EE', color: '#085041' };

        return { label: '終了', bg: '#F1EFE8', color: '#5F5E5A' };

      }

      function avatar(name, size = 24) {

        return `<div class="avatar" style="width:${size}px;height:${size}px;background:${

          MEMBER_BG[name] || '#F1EFE8'

        };color:${MEMBER_COLOR[name] || '#5F5E5A'};font-size:${

          size * 0.38

        }px">${name === '両方' ? '両' : name[0]}</div>`;

      }

      function badge(label, bg, color) {

        return `<span class="badge" style="background:${bg};color:${color}">${label}</span>`;

      }

      function progBar(pct) {

        const c = pct > 100 ? '#E24B4A' : pct > 80 ? '#EF9F27' : '#1D9E75';

        return `<div class="prog-bar"><div class="prog-fill" style="width:${Math.min(

          pct,

          100

        )}%;background:${c}"></div></div>`;

      }

      function chipRow(options, selectedKey, inputId, labelFn) {

        return options

      window.createTrip = function () {

        const start = document.getElementById('nt-start').value;

        const end = document.getElementById('nt-end').value;

          return;

        while (d <= ed) {

            date:

              d.getMonth() +

              1 +

              '/' +

              d.getDate() +

              '(' +

              DOW[d.getDay()] +

              ')',

            label: '',

          });

          d.setDate(d.getDate() + 1);

        }

        const trip = {

          id: 'trip' + Date.now(),

          name,

          cover,

          startDate: start,

          endDate: end,

          members: ['ケンケン', 'あき'],

          flights: [],

          hotels: [],

          places: [],

          expenses: [],

          budgets: CATS.map((cat) => ({ cat, budgetJPY: 0, note: '' })),

          schedule: [],

          checklist: [],

          days,

        };

        saveTrip(trip);

        window.trips.push(trip);

        renderTripList();

        hideModal('newTripModal');

        document.getElementById('nt-name').value = '';

        document.getElementById('nt-start').value = '';

        document.getElementById('nt-end').value = '';

      };

      function renderChecklist(t) {

            : items.filter((i) => i.cat === f);

          ? Math.round((checked / items.length) * 100)

          : 0;
